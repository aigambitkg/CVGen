Code

Simple, expressive
Python API

hello_quantum.py 
from cvgen import QuantumCircuit
from cvgen.backends.simulator import StateVectorSimulator

# Build a Bell state circuit
qc = QuantumCircuit(2)
qc.h(0)          # Hadamard on qubit 0
qc.cx(0, 1)     # CNOT: entangle qubits
qc.measure_all()

# Execute
sim = StateVectorSimulator()
result = sim.execute(qc)
print(result.counts)       # {'00': ~500, '11': ~500}
print(result.most_likely()) # '00' or '11'
quantum_search.py 
from cvgen.agents.quantum_agent import QuantumAgent, SearchTask
from cvgen.backends.simulator import StateVectorSimulator

# Define what to search for
task = SearchTask(
    num_qubits=4,
    target_states=[7, 11],  # Find states |0111⟩ and |1011⟩
    oracle_fn=lambda x: x in [7, 11]
)

# Agent handles everything: superposition, oracle, diffusion
agent = QuantumAgent(backend=StateVectorSimulator())
result = agent.run(task)

print(result.solution)   # 7 or 11 (found in √N steps)
print(result.confidence) # ~0.97
molecular_sim.py 
from cvgen.agents.hybrid_agent import HybridAgent, VariationalTask
from cvgen.backends.simulator import StateVectorSimulator

# Define a molecular Hamiltonian
task = VariationalTask(
    num_qubits=4,
    cost_observable={"ZIZI": 0.5, "IZIZ": -0.3, "XXII": 0.2},
    ansatz_depth=3,
    max_iterations=200
)

# Hybrid quantum-classical optimization
agent = HybridAgent(backend=StateVectorSimulator())
result = agent.run(task)

print(f"Ground state energy: {result.optimal_value:.6f}")
print(f"Converged in {result.iterations} iterations")
curl commands 
# Start the API server
$ uvicorn cvgen.api.app:app --reload

# Execute a circuit via REST
$ curl -X POST http://localhost:8000/api/v1/circuits/execute \
  -H "Content-Type: application/json" \
  -d '{
    "num_qubits": 2,
    "operations": [
      {"gate": "h", "targets": [0]},
      {"gate": "cx", "targets": [0, 1]},
      {"gate": "measure", "targets": [0, 1]}
    ],
    "shots": 1024
  }'

# List available backends
$ curl http://localhost:8000/api/v1/backends

# Run Grover search via API
$ curl -X POST http://localhost:8000/api/v1/agents/grover \
  -d '{"num_qubits": 3, "target_states": [5]}'

Quick Start

Up and running
in 60 seconds

1

Install

pip install cvgen

Core library with built-in simulator. No external dependencies needed.

2

Build a circuit

qc = QuantumCircuit(2)
qc.h(0).cx(0, 1).measure_all()

Method chaining for clean, readable circuit construction.

3

Execute

result = StateVectorSimulator().execute(qc)
print(result.counts)

Instant results. Probabilities, state vectors, measurement counts.

4

Scale up

pip install cvgen[api,ibm,braket]
uvicorn cvgen.api.app:app

Add REST API, cloud backends, web UI. Same code, real quantum hardware.
